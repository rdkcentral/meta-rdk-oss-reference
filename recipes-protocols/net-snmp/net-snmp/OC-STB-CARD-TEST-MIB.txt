-- Sample table

ocStbCardTable OBJECT-TYPE
    SYNTAX      SEQUENCE OF OcStbCardEntry
    MAX-ACCESS  not-accessible
    STATUS      current
    DESCRIPTION
            "This table used for table walks as well as Host 
             translation of GetBulkRequest into GetNextRequests.
             It may also be used to incur an endOfMibView condition"
		::= { ocStbCardTree 5 }

ocStbCardEntry OBJECT-TYPE
    SYNTAX      OcStbCardEntry
    MAX-ACCESS  not-accessible
    STATUS      current
    DESCRIPTION
            "Entry definition."
    INDEX { ocStbCardIndex }
    ::= { ocStbCardTable 1 }

OcStbCardEntry ::=
    SEQUENCE {
        ocStbCardIndex
            Unsigned32,
        ocStbCardTableType
            Unsigned32,
        ocStbCardTableDesc
            SnmpAdminString,
        ocStbCardTableStatus
            INTEGER
     }

ocStbCardIndex OBJECT-TYPE
    SYNTAX      Unsigned32 (1..4294967295)
    MAX-ACCESS  not-accessible
    STATUS      current
    DESCRIPTION
            "Index of the table."
    ::= { ocStbCardEntry 1 }

ocStbCardTableType OBJECT-TYPE
    SYNTAX      Unsigned32 (1..4)
    MAX-ACCESS  read-only
    STATUS      current
    DESCRIPTION
            "Creating 4 conceptual rows.  First
             row should have type 1, second row type 2
             etc, up to four rows."
    ::= { ocStbCardEntry 2 }

ocStbCardTableDesc OBJECT-TYPE
    SYNTAX      SnmpAdminString
    MAX-ACCESS  read-only
    STATUS      current
    DESCRIPTION
            "Textual description for each row.
             Replace n with number of row."
    DEFVAL			{ "Row n" }
    ::= { ocStbCardEntry 3 }

ocStbCardTableStatus OBJECT-TYPE
    SYNTAX      INTEGER
                {
                enabled(1),
                disabled(2)
                }
    MAX-ACCESS  read-only
    STATUS      current
    DESCRIPTION
            "Status of rows.  First two rows should be enabled.  Last
             two rows should be disabled."
    ::= { ocStbCardEntry 4 }
